获取命令返回值 获取命令输出内容

# 获取命令 echo 返回结果:
echo "HelloWorld"
if [ "$?" == 0 ]
then
     echo "return value is 0"
else
     echo "return value is not 0! return="$?
fi

# 获取命令 echo 输出内容
outValue=`echo "HelloWorld"`
if [ "$outValue" == "HelloWorld" ]
then
    echo "outValue is ->"$outValue
else
    echo "outValue value is not HelloWorld! outValue="$outValue
fi
